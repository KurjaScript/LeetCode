### 方法 1

- 复杂度：
    - 时间 O()
    - 空间 O()

``` js
var firstUniqChar = function(s) {
    for(v of s){
        if(s.indexOf(v) === s.lastIndexOf(v)){
            return s.indexOf(v)
        }
    }
    return -1
};
```
