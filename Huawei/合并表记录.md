### 方法 1

- 复杂度：
  - 时间 O()
  - 空间 O()

``` js
let len = readline();
let line;
let obj = {};
let i = 0;
while(true){
    line = readline()
    if(i < Number(len)){
        const arr = line.split('');
        const key = parseInt(arr[0]);
        const value = parseInt(arr[1])
        if(obj.hasOwnProperty(key)){
            obj[key] = obj[key] + value
        }else{
            obj[key] = value || 0;
        }
        i++;
    }else{
        break;
    }
}

for(const key in obj){
    console.log(key + ' ' + obj[key])
}
```
